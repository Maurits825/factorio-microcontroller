VAR DELTA
#9 instructions in between input reads, 9*1.33 = 12 seconds in between
MOVLW 0d12
MOVWF DELTA

VAR current_value
VAR new_value
VAR rate

RINF,1 current_value

loop
RINF,1 new_value

MOVFW new_value
SUBWF,w current_value

DIVWF,w DELTA
MOVWF rate
WOUTF,2 rate

MOVFW new_value
MOVWF current_value

GOTO loop
